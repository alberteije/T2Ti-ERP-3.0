CREATE
VIEW VIEW_SPED_NFE_DESTINATARIO
AS

select 
	DISTINCT(NFE.ID_CLIENTE) AS ID,
	D.NOME,
	D.CPF,
	D.CNPJ,
	D.INSCRICAO_ESTADUAL,
	D.CODIGO_MUNICIPIO,
	D.SUFRAMA,
	D.LOGRADOURO,
	D.NUMERO,
	D.COMPLEMENTO,
	D.BAIRRO
from
	NFE_CABECALHO NFE 
	INNER JOIN NFE_DESTINATARIO D ON (D.ID_NFE_CABECALHO = NFE.ID)
where 
	NFE.ID_CLIENTE IS NOT NULL
