	CREATE
	VIEW t2tierp.VIEW_SPED_I155
	AS
	(
	select 
		DATE_FORMAT(LC.DATA_LANCAMENTO, '%m/%Y') AS MES_ANO,
		LD.ID_CONTABIL_CONTA, 
		C.CLASSIFICACAO,
		LD.TIPO, 
		SUM(LD.VALOR) AS SOMA_VALOR
	from 
		CONTABIL_LANCAMENTO_DETALHE LD
		INNER JOIN CONTABIL_LANCAMENTO_CABECALHO LC ON (LD.ID_CONTABIL_LANCAMENTO_CAB = LC.ID)
		INNER JOIN CONTABIL_CONTA C ON (LD.ID_CONTABIL_CONTA = C.ID)
	group by 
		ID_CONTABIL_CONTA, TIPO, CLASSIFICACAO, MES_ANO
	)
